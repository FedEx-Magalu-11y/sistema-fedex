async function atualizarStatus(code) { const novoStatus = prompt("Digite o novo status (Ex: Em trânsito, Saiu para entrega, Entregue)"); if (!novoStatus) { alert("Atualização cancelada"); return; } const localizacao = prompt("Digite a localização (Ex: Centro Logístico SP)"); const { data: tracking, error: fetchError } = await supabase .from("trackings_test") .select("timeline") .eq("code", code) .single(); if (fetchError) { alert("Erro ao buscar dados"); return; } let timeline = tracking.timeline || []; timeline.push({ date: new Date().toISOString(), status: novoStatus, location: localizacao || "Centro Logístico" }); const { error } = await supabase .from("trackings_test") .update({ status: novoStatus, timeline: timeline, last_update: new Date().toISOString() }) .eq("code", code); if (error) { alert("Erro ao atualizar status"); } else { alert("Status atualizado com sucesso"); location.reload(); } }
